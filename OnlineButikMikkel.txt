//// -- LEVEL 1
//// -- Tables and References

// Creating tables
Table Customers {
  customer_id varchar [pk]
  phone int
  email varchar
  full_name varchar
  address varchar
  transaction_id int
}


Table Orders {
  order_id varchar [pk]
  customer_id varchar [ref: - Customers.customer_id]
  order_date time
  shipped_date time
 }
 
 
Table Items {
  order_id varchar [pk, ref: - Orders.order_id]
  item_id varchar [pk]
  product_id varchar
  quantity int [ref: - Stocks.quantity]
  price float
  discount int
  checkout boolean
}

Table Products {
  product_id varchar [pk, ref: - Items.product_id]
  category_id varchar
  brand_id varchar
  sex_id varchar
  name char
  price float
  weight float
  product_date time
}

Table Deal{
  deal_id varchar [pk, ref: - Products.product_id]
  deal_discount varchar
  deal_start date
  deal_end date
}

Table Sex{
  Sex_id varchar [pk, ref: - Products.sex_id]
  Sex_name varchar
}

Table Brands{
  brand_id varchar [pk, ref: - Products.brand_id]
  brand_name varchar
}

Table Stocks {
  product_id varchar [pk, ref: - Products.product_id]
  quantity int
}

Table Categories {
  category_id varchar [pk, ref: - Products.category_id]
  category varchar
}
 

Ref: "Products"."product_date" < "Products"."product_date"
